Evolution of central abundances¶

He destruction and C production via triple-alpha reaction.

In [9]:
plt.figure(1)
cyclerange=range(1500,2020,20)
he4=sefiles.get(cyclerange,'He-4')
c12=sefiles.get(cyclerange,'C-12')
n14=sefiles.get(cyclerange,'N-14')

he4_c=[]
c12_c=[]
n14_c=[]
for k in range(len(cyclerange)):
        he4_c.append(he4[k][0])
        c12_c.append(c12[k][0])
        n14_c.append(n14[k][0])

markevery=50
plt.plot(cyclerange,he4_c,label='He-4',marker='s',markevery=markevery)
plt.plot(cyclerange,c12_c,label='C-12',marker='^',markevery=markevery)
plt.plot(cyclerange,n14_c,label='N-14',marker='>',markevery=markevery)
plt.yscale('log')
plt.xlabel('Cycle number')
plt.ylabel('log-scaled X$_i$')
plt.legend(loc=2)
plt.title('Abundance evolution in center')
